  \def\thehour{#1#2}%
  \def\theminute{#3#4}%
  \def\thesecond{#5#6}%

  {%
    \def\thetimezonehour{+00}%
    \def\thetimezoneminute{00}%
  }%
  {%
    \parsepdftimezone#7%
  }%
}
